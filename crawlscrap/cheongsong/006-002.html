006-002. using selenium and BeautifulSoup (2) @ from selenium import webdriver import time from bs4 import BeautifulSoup browser = webdriver.Chrome("/users/username/chromedriver") browser.get("https://www.w3school.com/colors/colors_rgb.asp") time.sleep(2) # I remove existing value browser.find_elements_by_css_selector("#r01").clear() browser.find_elements_by_css_selector("#r01").send_keys("1") browser.find_elements_by_css_selector("#g01").clear() browser.find_elements_by_css_selector("#g01").send_keys("20") browser.find_elements_by_css_selector("#b01").clear() browser.find_elements_by_css_selector("#b01").send_keys("44") html = browser.page_source soup = BeautifulSoup(html, 'html.parser') color = soup.find("div", {"id":"hex01"}).text print(color) time.sleep(10) browser.quit()
